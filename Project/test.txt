
int main() {
char GRADE = 'B';

switch(GRADE)
  {
    case 'A' :
       cout << "Excellent!" <<;
       break;
    case 'B' :
       cout << "Pfffft Average!" <<;
    case 'C' :
       cout << "Well done" <<;
       break;
    case 'D' :
       cout << "You passed" <<;
       break;
    case 'F' :
       cout << "Better try again" <<;
       break;
    default :
       cout << "Invalid grade" <<;
  }
cout << "Your grade is " <<;

cout<< GRADE <<;

}
